### 접근
제약이 크지 않아서 간단하게 sort 함수를 사용했다.

### 코드
```c++
#include <iostream>
#include <algorithm>

using namespace std ;

bool
compare (int a, int b)
{
    return a > b ;
}

int
main ()
{
    int tc_cnt ;
    cin >> tc_cnt ;

    int ** tc_arr = new int * [tc_cnt] ;    
    for (int tc = 0; tc < tc_cnt; tc++) {
        tc_arr[tc] = new int[10] ;
        for (int i = 0; i < 10; i++) cin >> tc_arr[tc][i] ;
        sort(tc_arr[tc], tc_arr[tc] + 10, compare) ;
    }

    for (int tc = 0; tc < tc_cnt; tc++) {
        cout << tc_arr[tc][2] << endl ;
        delete[] tc_arr[tc] ;
    }
    delete[] tc_arr ;

    return 0 ;
}
```
